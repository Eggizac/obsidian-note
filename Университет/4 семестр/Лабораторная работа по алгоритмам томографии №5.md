``` python
import numpy as np

import matplotlib.pyplot as plt

import time

  

def build_full_rank_A(M):

n = M * M

A = np.zeros((n, n))

rays_found = 0

np.random.seed(42)

while rays_found < n:

theta = np.random.uniform(0, np.pi)

r = np.random.uniform(-M*0.8, M*0.8)

cos_t, sin_t = np.cos(theta), np.sin(theta)

t_samples = 300 * M

t = np.linspace(-M*1.5, M*1.5, t_samples)

x = r * cos_t - t * sin_t

y = r * sin_t + t * cos_t

c = np.floor(x + M/2).astype(int)

r_idx = np.floor(M/2 - y).astype(int)

valid = (c >= 0) & (c < M) & (r_idx >= 0) & (r_idx < M)

c, r_idx = c[valid], r_idx[valid]

row = np.zeros(n)

dt = (3 * M) / t_samples

for r_i, c_i in zip(r_idx, c):

row[r_i * M + c_i] += dt

if np.max(row) > 0:

A[rays_found] = row

if np.linalg.matrix_rank(A[:rays_found+1]) == rays_found + 1:

rays_found += 1

else:

A[rays_found] = 0

return A

  

def run_experiment(M):

n = M * M

# Генерация фантома

x_true = np.random.rand(n)

# Построение системы уравнений

t0 = time.time()

A = build_full_rank_A(M)

t_build = time.time() - t0

cond_A = np.linalg.cond(A)

# Вычисление идеальных проекций

p = A @ x_true

# Решение системы (матричное обращение)

t0 = time.time()

A_inv = np.linalg.inv(A)

x_rec = A_inv @ p

t_inv = time.time() - t0

# Оценка качества реконструкции

mse = np.mean((x_rec - x_true)**2)

print(f"Размер {M}x{M} (n={n}) | T_build={t_build:.4f}с | T_inv={t_inv:.5f}с | k(A)={cond_A:.1e} | MSE={mse:.2e}")

return x_true.reshape(M, M), x_rec.reshape(M, M)

  

img_true_2, img_rec_2 = run_experiment(M=2)

img_true_3, img_rec_3 = run_experiment(M=3)

img_true_4, img_rec_4 = run_experiment(M=4)

  

fig, axes = plt.subplots(2, 3, figsize=(10, 6))

  

axes[0, 0].imshow(img_true_2, cmap='gray'); axes[0, 0].set_title("1. Исходный 2x2")

axes[0, 1].imshow(img_true_3, cmap='gray'); axes[0, 1].set_title("2. Исходный 3x3")

axes[0, 2].imshow(img_true_4, cmap='gray'); axes[0, 2].set_title("3. Исходный 4x4")

  

axes[1, 0].imshow(img_rec_2, cmap='gray'); axes[1, 0].set_title("4. Восстановлен 2x2")

axes[1, 1].imshow(img_rec_3, cmap='gray'); axes[1, 1].set_title("5. Восстановлен 3x3")

axes[1, 2].imshow(img_rec_4, cmap='gray'); axes[1, 2].set_title("6. Восстановлен 4x4")

  

plt.tight_layout()

plt.show()
```
