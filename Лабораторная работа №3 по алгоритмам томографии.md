``` python
import numpy as np

import matplotlib.pyplot as plt

import scipy.io as sio

  

N = 128

delta = 0.5

num_angles = 180

num_det = 180

  

# Подготовка фантома и синусограммы

phantom = np.zeros((N, N))

x_coords = (np.arange(N) - (N - 1) / 2) * delta

y_coords = ((N - 1) / 2 - np.arange(N)) * delta

X, Y = np.meshgrid(x_coords, y_coords)

  

circles = [

{'mu': 1.0, 'r': 20, 'x0': 0, 'y0': 0},

{'mu': 0.5, 'r': 10, 'x0': -15, 'y0': 15},

{'mu': 0.8, 'r': 8, 'x0': 20, 'y0': -10}

]

for c in circles:

mask = (X - c['x0'])**2 + (Y - c['y0'])**2 <= c['r']**2

phantom[mask] += c['mu']

  

# Генерация синусограммы (Прямое преобразование Радона)

angles = np.linspace(0, np.pi, num_angles, endpoint=False)

xi_max = (np.sqrt(2) / 2) * (N * delta)

detector_pos = np.linspace(-xi_max, xi_max, num_det)

sinogram = np.zeros((num_angles, num_det))

  

for k, theta in enumerate(angles):

cos_t, sin_t = np.cos(theta), np.sin(theta)

xi_grid = X * cos_t + Y * sin_t

d_xi = detector_pos[1] - detector_pos[0]

idx = np.round((xi_grid - detector_pos[0]) / d_xi).astype(int)

valid = (idx >= 0) & (idx < num_det)

for j in range(num_det):

mask_det = (idx == j) & valid

sinogram[k, j] = np.sum(phantom[mask_det])

  

# Восстановление

print("Вычисление суммарного изображения...")

g = np.zeros((N, N))

d_xi = detector_pos[1] - detector_pos[0]

  

for k, theta in enumerate(angles):

xi_grid = X * np.cos(theta) + Y * np.sin(theta)

idx = np.round((xi_grid - detector_pos[0]) / d_xi).astype(int)

valid = (idx >= 0) & (idx < num_det)

g[valid] += sinogram[k, idx[valid]]

g /= num_angles

  

# Пункт 3: Двумерное БПФ

print("Частотная фильтрация...")

G = np.fft.fftshift(np.fft.fft2(g))

  

# Пункт 4: Применение фильтра rho/2 и аподизации (Окно Хэмминга)

d_rho = 1.0 / (N * delta)

rho_1d = (np.arange(N) - N / 2) * d_rho

rho_x, rho_y = np.meshgrid(rho_1d, rho_1d)

rho = np.sqrt(rho_x**2 + rho_y**2)

  

A_rho = 0.54 + 0.46 * np.cos(np.pi * rho / np.max(rho)) # Окно Хэмминга

filtered_spec = G * (rho / 2.0) * A_rho

  

# Пункт 5: Обратное БПФ

reconstructed = np.real(np.fft.ifft2(np.fft.ifftshift(filtered_spec)))

  

# Пункт 6: Обрезка краевых артефактов (по 15 пикселей)

crop = 15

final_mu = reconstructed[crop:-crop, crop:-crop]

final_x = x_coords[crop:-crop]

final_y = y_coords[crop:-crop]

  

sio.savemat('reconstructed.mat', {

'mu': final_mu,

'x': final_x,

'y': final_y

})

  

fig, axes = plt.subplots(2, 2, figsize=(10, 10))

axes[0, 0].imshow(phantom, cmap='bone'); axes[0, 0].set_title("1. Исходный фантом")

axes[0, 1].imshow(sinogram, aspect='auto', cmap='jet'); axes[0, 1].set_title("2. Синограмма")

axes[1, 0].imshow(g, cmap='bone'); axes[1, 0].set_title("3. Суммарное изобр. (без фильтра)")

axes[1, 1].imshow(reconstructed, cmap='bone'); axes[1, 1].set_title("4. Результат восстановления")

plt.tight_layout()

plt.show()
```
