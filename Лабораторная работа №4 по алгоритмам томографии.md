# Код
``` python
import numpy as np

import matplotlib.pyplot as plt

import scipy.io as sio

  

N = 128

delta = 0.5

num_angles = 180

num_det = 180

  

phantom = np.zeros((N, N))

x_coords = (np.arange(N) - (N - 1) / 2) * delta

y_coords = ((N - 1) / 2 - np.arange(N)) * delta

X, Y = np.meshgrid(x_coords, y_coords)

  

circles = [

{'mu': 1.0, 'r': 20, 'x0': 0, 'y0': 0},

{'mu': 0.5, 'r': 10, 'x0': -15, 'y0': 15},

{'mu': 0.8, 'r': 8, 'x0': 20, 'y0': -10}

]

for c in circles:

mask = (X - c['x0'])**2 + (Y - c['y0'])**2 <= c['r']**2

phantom[mask] += c['mu']

  

angles = np.linspace(0, np.pi, num_angles, endpoint=False)

L_phys = N * delta

xi_max = (np.sqrt(2) / 2) * L_phys

detector_pos = np.linspace(-xi_max, xi_max, num_det)

sinogram = np.zeros((num_angles, num_det))

  

# Прямое проецирование

for k, theta in enumerate(angles):

cos_t, sin_t = np.cos(theta), np.sin(theta)

xi_grid = X * cos_t + Y * sin_t

d_xi = detector_pos[1] - detector_pos[0]

idx = np.round((xi_grid - detector_pos[0]) / d_xi).astype(int)

valid = (idx >= 0) & (idx < num_det)

for j in range(num_det):

mask_det = (idx == j) & valid

sinogram[k, j] = np.sum(phantom[mask_det])

  

# Алгоритм FBP

print("Начало фильтрации проекций...")

filtered_sinogram = np.zeros_like(sinogram)

d_xi = detector_pos[1] - detector_pos[0]

  

# Формирование частотной сетки

d_omega = (2 * np.pi) / (num_det * d_xi)

# Используем стандартный сдвиг частот для БПФ

omega = np.fft.fftfreq(num_det, d=d_xi) * 2 * np.pi

omega_cut = np.pi / d_xi

  

# Аподизирующая функция (квадрат косинуса)

A_omega = np.zeros_like(omega)

valid_omega = np.abs(omega) <= omega_cut

A_omega[valid_omega] = np.cos((np.pi * np.abs(omega[valid_omega])) / (2 * omega_cut))**2

  

H_omega = np.abs(omega) / np.sqrt(2 * np.pi)

  

# Шаг 2: Фильтрация каждой проекции в частотной области

for k in range(num_angles):

p = sinogram[k, :]

P_omega = np.fft.fft(p)

P_filtered = P_omega * H_omega * A_omega

# Обратное БПФ

p_filtered = np.real(np.fft.ifft(P_filtered))

filtered_sinogram[k, :] = p_filtered

  

print("Выполнение обратной проекции...")

# Шаг 3: Обратная проекция отфильтрованных проекций

mu_reconstructed = np.zeros((N, N))

  

for k, theta in enumerate(angles):

xi_grid = X * np.cos(theta) + Y * np.sin(theta)

# Интерполяция по ближайшему соседу

idx = np.round((xi_grid - detector_pos[0]) / d_xi).astype(int)

valid = (idx >= 0) & (idx < num_det)

mu_reconstructed[valid] += filtered_sinogram[k, idx[valid]]

  

# Нормировка

mu_reconstructed = mu_reconstructed * (np.pi / num_angles)

  

# Шаг 4: Обрезка краевых артефактов

crop = 15

mu_final = mu_reconstructed[crop:-crop, crop:-crop]

x_final = x_coords[crop:-crop]

y_final = y_coords[crop:-crop]

  

# Шаг 6: Сохранение результата

sio.savemat('reconstructed_fbp.mat', {

'mu': mu_final,

'x': x_final,

'y': y_final

})

print("Результат сохранен в файл 'reconstructed_fbp.mat'.")

  

# Шаг 5: Визуализация

fig, axes = plt.subplots(1, 2, figsize=(12, 5))

  

im1 = axes[0].imshow(phantom, extent=[x_coords[0], x_coords[-1], y_coords[-1], y_coords[0]], cmap='bone')

axes[0].set_title("Исходный фантом")

axes[0].set_xlabel("x, см"); axes[0].set_ylabel("y, см")

fig.colorbar(im1, ax=axes[0])

  

im2 = axes[1].imshow(mu_final, extent=[x_final[0], x_final[-1], y_final[-1], y_final[0]], cmap='bone')

axes[1].set_title("Реконструкция FBP (с обрезкой артефактов)")

axes[1].set_xlabel("x, см"); axes[1].set_ylabel("y, см")

fig.colorbar(im2, ax=axes[1])

  

plt.tight_layout()

plt.show()
```
